Mysql e Postgresql
------------------

CREATE DATABASE pmdb;

Mysql
-----
CREATE TABLE instituicao (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    cidade VARCHAR(255),
    estado VARCHAR(2)
);

Postgresql
----------

CREATE TABLE instituicao (
    id SERIAL PRIMARY KEY,
    nome VARCHAR(255) NOT NULL,
    cidade VARCHAR(255),
    estado CHAR(2)
);

MongoDB
-------
use pmdb;

db.createCollection("instituicao")
